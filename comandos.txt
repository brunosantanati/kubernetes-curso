- Subir o container baseado no arquivo docker-compose.yaml:

sudo docker-compose up -d
sudo docker-compose up --force-recreate db

- Ver containers rodando:

sudo docker ps

- Matar container

sudo docker kill projeto_web_1

- Criar imagem customizada:

sudo docker build -t "aplicacao-loja:v1" .

- Criar tabelas no container do banco:

sudo docker exec -it projeto_db_1 sh
mysql -u root
use loja
create table produtos (id integer auto_increment primary key, nome varchar(255), preco decimal(10,2));
alter table produtos add column usado boolean default false;
alter table produtos add column descricao varchar(255);
create table categorias (id integer auto_increment primary key, nome varchar(255));
insert into categorias (nome) values ("Futebol"), ("Volei"), ("Tenis");
alter table produtos add column categoria_id integer;
update produtos set categoria_id = 1;

http://localhost:8080
